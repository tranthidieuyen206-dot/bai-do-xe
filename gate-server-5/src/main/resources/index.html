const form = document.getElementById('carForm'); const submitBtn = document.getElementById('submitBtn'); const inputBienSo = document.getElementById('bienSo'); const logList = document.getElementById('logList'); let isFirstLog = true; // ====================================================== // THUẬT TOÁN SẮP XẾP HÀNG ĐỢI (ROUND-ROBIN) TẠI CLIENT // ====================================================== // Điền link Render của cả 3 cổng vào đây const danhSachCong = [ "https://gate-server-2.onrender.com", "https://gate-server-4.onrender.com", "https://gate-server-5.onrender.com" ]; // Biến nhớ xem đang đến lượt cổng nào tiếp khách let luotTiepTheo = 0; form.addEventListener('submit', async function(e) { e.preventDefault(); const bienSo = inputBienSo.value.trim().toUpperCase(); if(!bienSo) return; submitBtn.disabled = true; submitBtn.innerHTML = '⏳ Đang phân luồng...'; // 1. CHỌN CỔNG ĐỂ GIAO VIỆC const congDuocChon = danhSachCong[luotTiepTheo]; // 2. TĂNG LƯỢT CHO LẦN SAU (Nếu đến cuối danh sách thì quay lại cổng số 0) luotTiepTheo = (luotTiepTheo + 1) % danhSachCong.length; try { // 3. GỬI XE THẲNG ĐẾN CỔNG VỪA ĐƯỢC CHỌN const apiUrl = congDuocChon + '/api/nhan-xe'; const response = await fetch(apiUrl, { method: 'POST', body: bienSo }); if (response.ok) { const resultText = await response.text(); addLog("✅ [Phân luồng tới " + congDuocChon.split('-')[2] + "] " + resultText); inputBienSo.value = ''; } else { addLog('❌ Lỗi xử lý từ cổng được phân công!'); } } catch (error) { addLog('❌ Mất kết nối! Cổng ' + congDuocChon + ' có thể đang sập!'); console.error(error); } finally { submitBtn.disabled = false; submitBtn.innerHTML = 'Cho Xe Vào Bãi'; inputBienSo.focus(); } }); // Hàm thêm dòng lịch sử lên màn hình (Giữ nguyên của bạn) function addLog(message) { if (isFirstLog) { logList.innerHTML = ''; isFirstLog = false; } const now = new Date(); const timeString = now.toLocaleTimeString('vi-VN'); const li = document.createElement('li'); li.className = 'log-item'; li.innerHTML = `[${timeString}] ${message}`; logList.insertBefore(li, logList.firstChild); }
